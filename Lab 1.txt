#include <iostream>
using namespace std;

void checkOperators(const string& input) {

    cout << "Operators in the input: ";
    bool foundOperator = false;
    for (char ch : input) {
        if (ch == '+' || ch == '-' || ch == '*' || ch == '/' || ch == '%' || ch == '=') {
            cout << ch << " ";
            foundOperator = true;
        }
    }

    if (!foundOperator) {
        cout << "No operators found";
    }
    cout << endl;
}

int main() {
    string input;

    cout << "Enter an expression: ";
    cin >> input;

    checkOperators(input);

    return 0;
}
